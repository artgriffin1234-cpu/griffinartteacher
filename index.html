function doGet(e){ var ss = SpreadsheetApp.getActiveSpreadsheet(); if(e.parameter.action === 'ping'){ return ContentService.createTextOutput('OK-DELETE-V3'); } if(e.parameter.action === 'read'){ var tz = ss.getSpreadsheetTimeZone(); var staffSheet = ss.getSheetByName('staff'); var attSheet = ss.getSheetByName('attendance'); var staffData = staffSheet ? staffSheet.getDataRange().getValues() : []; var attData = attSheet ? attSheet.getDataRange().getValues() : []; staffData.forEach(function(row, i){ if(i === 0) return; row.forEach(function(cell, c){ if(cell instanceof Date){ row[c] = Utilities.formatDate(cell, tz, 'yyyy-MM-dd'); } }); }); attData.forEach(function(row, i){ if(i === 0) return; if(row[1] instanceof Date){ row[1] = Utilities.formatDate(row[1], tz, 'yyyy-MM-dd'); } }); var payload = JSON.stringify({staff:staffData, att:attData}); var cb = e.parameter.callback; if(cb){ return ContentService.createTextOutput(cb + '(' + payload + ')').setMimeType(ContentService.MimeType.JAVASCRIPT); return sh; } var hdr = sh.getRange(1, 1, 1, sh.getLastColumn()).getValues()[0]; if(header.length > hdr.length){ sh.getRange(1, 1, 1, header.length).setValues([header]); } return sh; } function upsertStaffRow(ss, r){ var sh = ensureSheet(ss, 'staff', ['id','name','role','cls','sub','wage','sid','phone','account','birth','region','year','months','tenureAnchor']);
